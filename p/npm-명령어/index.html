npm 명령어

Jan 15, 2022
1 minute read 
# npm 프로젝트 시작 ( package.json 생성됨 )
# -y 를 붙이면 init 초기화시 기본설정을 물어보지 않고 기본으로 설치된다.
npm init -y                    

# 서버구동
npx serve

# 개발용 의존성 (devDependencies) 으로 parcel-bundler 설치
npm install parcel-bundler -D  

# 일반 의존성 (dependencies) 으로 lodash 패키지(모듈 modules) 설치
npm install lodash             

# node_modules 폴더 삭제시 package.json 에 설정된 모듈 재설치
npm install [혹은 npm i]   

# 기존 패키지의 다른 버전으로 설치된 버전을 덮어 씌우며 설치함   
npm install lodash@4.17.20     

# 패키지를 업데이트 (단 json 파일 패키지 버전에 ^ 이 없으면 업데이트 안됨)
npm update lodash              

# package.json 파일 내부 script 에 설정한 dev 라는 이름의 스크립트 명령어가 실행됨
npm run dev                    

# 패키지 lodash 의 정보를 확인함
npm info lodash 

# npm cache 삭제
npm cache clean --force    

# npm cache 무결성 검사 및 가비지 데이터 삭제
npm cache verify           

# 설치된 패키지 조회
npm ls
npm CLI
